package org.pdfsam.model.tool;

/**
 * Values for a tool priorities
 *
 * @author Andrea Vacondio
 */
public enum ToolPriority {
    LOW(10),
    DEFAULT(0),
    HIGH(-10);
    private final int priority;

    ToolPriority(int priority) {
        this.priority = priority;
    }

    public int getPriority() {
        return priority;
    }

}
